cmake_minimum_required(VERSION 3.18)
project(PrintVariables_EX)

function(print_cmake_variables)
	get_cmake_property(varNames vars)
	list(SORT varNames)
	foreach(varNames ${varNames})
		if(ARGV0)
			unset(MATCHED)
			string(REGEX MATCH ${ARGV0} MATCHED ${varNames})
			if(NOT MATCHED)
				continue()
			endif()
		endif()
		message(STATUS "${varNames}${${varNames}}")
	endforeach()
endfunction(print_cmake_variables)

print_cmake_variables(CMAKE_CL)
